Number Pattern 1
Print the following pattern for the given N number of rows.
Pattern for N = 4
1
11
111
1111
Input format :
Integer N (Total no. of rows)
Output format :
Pattern in N lines
Sample Input :
5
Sample Output :
1
11
111
1111
11111


import java.util.Scanner;
public class Main {
	
	public static void main(String[] args) {
		// Write your code here
Scanner s = new Scanner(System.in);
		int a = s.nextInt();
		int i=1;
		while(i<=a)
		{
			int j=1;
			while(j<=i)
			{
				System.out.print("1");
				j+=1;
			}
			System.out.print("\n");
			i+=1;
		}
	}
}
