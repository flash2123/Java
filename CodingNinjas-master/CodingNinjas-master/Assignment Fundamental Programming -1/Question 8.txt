Number Pattern 2
Print the following pattern for the given N number of rows.
Pattern for N = 4
1
11
202
3003
Input format :
Integer N (Total no. of rows)
Output format :
Pattern in N lines
Sample Input :
5
Sample Output :
1
11
202
3003
40004


import java.util.Scanner;
public class Main {
	
	public static void main(String[] args) {
		// Write your code here
Scanner s = new Scanner(System.in);

		int n=s.nextInt();
		int i=1;
		int incr=1;
		while(i<=n){
			int j=1;
			while(j<=i){
				if(j==1){
					System.out.print(incr);
				}
				else if(j==i){
					System.out.print(incr);
					incr++;
				}
				else{
					System.out.print("0");
				}
				j++;
			}
			System.out.println("");
			i++;
		}


	}
}
